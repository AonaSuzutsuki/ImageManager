Examples

GetSha256 サンプル

readonly byte[] testData = Encoding.UTF8.GetBytes("Hello,World!!");
const string expectedHash = "899948C9C72B65EBB59A8D01EE4DB68BABCF91D55C6D93DE66637EA869A3EF1D";

public void GetSha256Test1()
{
    var hash = "";
    using (var stream = new MemoryStream(testData))
    {
        hash = Sha256.GetSha256(stream);
    }
    Assert.AreEqual(expectedHash, hash);
}

public void GetSha256Test2()
{
    var hash = Sha256.GetSha256(testData);
    Assert.AreEqual(expectedHash, hash);
}
